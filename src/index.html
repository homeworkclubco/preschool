Quick Start

Import components directly in your HTML:

<script type="module">
  import { HcModal } from '@hc/preschool';
</script>

<button data-modal="my-modal">Open Modal</button>

<hc-modal id="my-modal" label="Example">
  <p>Modal content here</p>
  <button data-close slot="footer">Close</button>
</hc-modal>
