6. Custom Time Range (10s - 30s)

Custom Time Range

Playing from 10s to 30s, then looping
Code:

  src="https://www.youtube.com/watch?v=..."
  start-at="10"
  end-at="30"
  autoplay
  muted
  loop
></hc-video-background>

8. JavaScript API

JavaScript API

Current Time: 0s

State: notstarted
Code: 
const video = document.getElementById('api-demo');

// Methods
video.play();
video.pause();
video.mute();
video.unmute();
video.setVolume(0.5);
video.seekTo(30);
video.seek(50); // 50%

// Properties
console.log(video.playing);
console.log(video.currentTime);
console.log(video.duration);
console.log(video.percentComplete);

// Events
video.addEventListener('hc-video-play', (e) => {
  console.log('Video playing!');
});

video.addEventListener('hc-video-time-update', (e) => {
  console.log('Current time:', e.detail.currentTime);
});
